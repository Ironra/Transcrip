### Dockerfile
```dockerfile
# Base runtime image
FROM mcr.microsoft.com/dotnet/aspnet:9.0 AS base
WORKDIR /app

# Exponer puertos HTTP y HTTPS que se usan en launchSettings.json
ENV ASPNETCORE_URLS="http://+:5142;https://+:7299"
EXPOSE 5142 7299

# Build image
FROM mcr.microsoft.com/dotnet/sdk:9.0 AS build
ARG BUILD_CONFIGURATION=Release
WORKDIR /src

# Copiar el .csproj y restaurar dependencias
COPY ["VoskRealtimeApi.csproj", "./"]
RUN dotnet restore "VoskRealtimeApi.csproj"

# Copiar el resto del código y compilar
COPY . .
RUN dotnet build "VoskRealtimeApi.csproj" -c $BUILD_CONFIGURATION -o /app/build

# Publicar la aplicación
FROM build AS publish
RUN dotnet publish "VoskRealtimeApi.csproj" -c $BUILD_CONFIGURATION -o /app/publish /p:UseAppHost=false

# Imagen final
FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "VoskRealtimeApi.dll"]
