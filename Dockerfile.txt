# 1) Build stage
FROM mcr.microsoft.com/dotnet/sdk:7.0 AS build
WORKDIR /src

# Copia csproj y restaura dependencias primero (cach√© de capas)
COPY ["VoskRealtimeApi.csproj", "./"]
RUN dotnet restore "./VoskRealtimeApi.csproj"

# Copia todo y publica en modo Release
COPY . .
RUN dotnet publish "VoskRealtimeApi.csproj" -c Release -o /app/publish

# 2) Runtime stage
FROM mcr.microsoft.com/dotnet/aspnet:7.0 AS runtime
WORKDIR /app

# Variables de entorno para que ASP.NET Core use el puerto que Fly asigna
ENV ASPNETCORE_URLS=http://*:${PORT:-80}

# Copia los artefactos publicados
COPY --from=build /app/publish .

# Exponer el puerto (no obligatorio: Fly lo asigna)
EXPOSE 80

# Punto de entrada
ENTRYPOINT ["dotnet", "VoskRealtimeApi.dll"]
